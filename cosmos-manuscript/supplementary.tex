\section{Supplementary Material}

\subsection{Variable definitions}

Target site
\begin{equation*}
    n \in \{1,\dots,N\}
\end{equation*}
%
Frame number
\begin{equation*}
    f \in \{1,\dots,F\}
\end{equation*}
%
$x$-axis pixel number
\begin{equation*}
    i \in \{1,\dots,I\}
\end{equation*}
%
$y$-axis pixel number
\begin{equation*}
    j \in \{1,\dots,J\}
\end{equation*}
%
Spot index
\begin{equation*}
    k \in \{1,\dots,K\}
\end{equation*}
%
Background intensity
\begin{equation*}
    b_{nf} > 0
\end{equation*}
%
Hidden state
\begin{equation*}
    m_{nf}
\end{equation*}
%
Integrated spot intensity
\begin{equation*}
    h_{nfk} > 0
\end{equation*}
%
Ideal spot shape
\begin{equation*}
    \mu^{D}_{nfij} = b_{nf} + \sum_k \dfrac{m_{nfk}h_{nfk}}{2 \pi w^2_{nfk}} \exp{\left ( -\dfrac{(i-x_{nfk})^2 + (j-y_{nfk})^2}{2w^2_{nfk}} \right)}
\end{equation*}
